# Templates Expressjs
 This is basic templates project expressjs for backend
# install:
    npm install
# setting .env:   
    change your environments in .env
    make new file .env.production if you want to deploy your project
# usage:
    npm run dev ( use .env.production)
    npm start ( use .env)
# Dependency
    Cors,
    Axios,
    dotenv,
    jsonwebtoken,
    nodemon,
    body-parser,
    socket.io,
    node-localstorage,
    mongodb,
    mongoose
